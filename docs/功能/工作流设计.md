# 节点式工作流系统设计

> 更新说明：若查看最新的 Job-WorkflowRun-NodeRun 模型和分类器节点设计，请优先阅读 [工作流设计（版本3）](../功能/工作流设计（版本3）.md)。

> 版本：v2.0 | 日期：2026-03-19
> 设计参考：ComfyUI 节点图模式

## 核心概念

工作流由**节点（Node）**和**连线（Edge）**组成有向无环图（DAG）。
每个节点代表一个处理操作，连线表示执行顺序和数据传递。
执行时按拓扑排序依次执行各节点，支持并行执行无依赖的分支。

```
示例：photo 工作流

  [触发器: photo 分类]
          |
          v
    [重命名节点]
    rule: 文件夹名+序号
          |
     +----+----+
     |         |
     v         v
[压缩节点]  [缩略图节点]  <- 并行执行
  ZIP格式   Emby规范
     |         |
     +----+----+
          |
          v
    [移动节点]
    target: /data/target/photos
```

## 节点类型

### 触发器节点（Trigger Node）

每个工作流有且仅有一个触发器节点，定义该工作流适用的文件夹分类。

```json
{
  "id": "trigger_01",
  "type": "trigger",
  "data": {
    "label": "分类触发器",
    "category": "photo"   // photo | video | mixed | manga | any
  },
  "position": { "x": 250, "y": 0 }
}
```

### 操作节点（Action Nodes）

#### rename — 批量重命名
```json
{
  "id": "rename_01",
  "type": "rename",
  "data": {
    "label": "重命名",
    "rule_id": "rule_uuid",
    "rule_name": "文件夹名+序号"
  }
}
```

#### compress — ZIP 压缩
```json
{
  "id": "compress_01",
  "type": "compress",
  "data": {
    "label": "ZIP 压缩",
    "output_dir": "",        // 留空=原地
    "delete_after": false
  }
}
```

#### thumbnail — Emby 缩略图
```json
{
  "id": "thumb_01",
  "type": "thumbnail",
  "data": {
    "label": "生成缩略图",
    "seek_offset": "00:10:00",
    "overwrite": false
  }
}
```

#### move — 移动文件夹
```json
{
  "id": "move_01",
  "type": "move",
  "data": {
    "label": "移动",
    "target_dir": "/data/target/photos"  // 留空=全局配置
  }
}
```

#### condition — 条件分支（Phase 3）
```json
{
  "id": "cond_01",
  "type": "condition",
  "data": {
    "label": "文件数量判断",
    "field": "image_count",
    "operator": "gt",
    "value": 10
  }
}
```

#### wait — 等待用户确认
```json
{
  "id": "wait_01",
  "type": "wait",
  "data": {
    "label": "等待确认",
    "message": "请确认重命名结果后继续"
  }
}
```

## 数据模型

```go
// 工作流（存储 React Flow 完整图 JSON）
type Workflow struct {
    ID        string          `json:"id" db:"id"`
    Category  string          `json:"category" db:"category"`
    Name      string          `json:"name" db:"name"`
    Graph     json.RawMessage `json:"graph" db:"graph"` // React Flow graph JSON
    CreatedAt time.Time       `json:"created_at" db:"created_at"`
    UpdatedAt time.Time       `json:"updated_at" db:"updated_at"`
}

// React Flow Graph JSON 结构
type WorkflowGraph struct {
    Nodes []WorkflowNode `json:"nodes"`
    Edges []WorkflowEdge `json:"edges"`
}

type WorkflowNode struct {
    ID       string          `json:"id"`
    Type     string          `json:"type"` // trigger|rename|compress|thumbnail|move|condition|wait
    Data     json.RawMessage `json:"data"`
    Position NodePosition    `json:"position"`
    Enabled  bool            `json:"enabled"`
}

type WorkflowEdge struct {
    ID     string `json:"id"`
    Source string `json:"source"`
    Target string `json:"target"`
    Label  string `json:"label,omitempty"` // 条件分支用："true" | "false"
}

type NodePosition struct {
    X float64 `json:"x"`
    Y float64 `json:"y"`
}
```

## 后端执行引擎

```go
// DAG 执行器
type WorkflowExecutor struct {
    graph    WorkflowGraph
    services *ServiceContainer
    broker   *sse.Broker
}

func (e *WorkflowExecutor) Execute(ctx context.Context, folderID string, jobID string) error {
    // 1. 拓扑排序
    order, err := e.topologicalSort()
    if err != nil {
        return fmt.Errorf("invalid DAG: %w", err)
    }

    // 2. 并行执行无依赖节点
    // 使用 errgroup + semaphore 控制并发
    for _, wave := range order {
        g, ctx := errgroup.WithContext(ctx)
        for _, nodeID := range wave {
            nodeID := nodeID
            g.Go(func() error {
                return e.executeNode(ctx, nodeID, folderID, jobID)
            })
        }
        if err := g.Wait(); err != nil {
            return err
        }
    }
    return nil
}

func (e *WorkflowExecutor) executeNode(ctx context.Context, nodeID, folderID, jobID string) error {
    node := e.findNode(nodeID)
    // 推送节点开始事件
    e.broker.Publish(sse.Event{Type: "workflow.node.start", Data: ...})

    var err error
    switch node.Type {
    case "rename":    err = e.services.Renamer.Execute(ctx, folderID, node.Data)
    case "compress":  err = e.services.Compressor.Execute(ctx, folderID, node.Data)
    case "thumbnail": err = e.services.Thumbnail.Execute(ctx, folderID, node.Data)
    case "move":      err = e.services.Mover.Execute(ctx, folderID, node.Data)
    case "wait":      err = e.handleWait(ctx, node)
    }

    // 推送节点完成/失败事件
    if err != nil {
        e.broker.Publish(sse.Event{Type: "workflow.node.failed", Data: ...})
    } else {
        e.broker.Publish(sse.Event{Type: "workflow.node.done", Data: ...})
    }
    return err
}
```

## API 设计

```
GET    /api/workflows                  # 所有工作流列表
GET    /api/workflows/:id              # 单个工作流（含完整 graph JSON）
POST   /api/workflows                  # 创建工作流
PUT    /api/workflows/:id              # 更新工作流（graph 全量替换）
DELETE /api/workflows/:id              # 删除工作流
POST   /api/workflows/:id/validate     # 校验 DAG（检测环路、孤立节点）
POST   /api/jobs/run-workflow          # 对 folder_ids 执行工作流
```

### 技术选型

- **React Flow** (`@xyflow/react`) — 节点图渲染、连线、缩放/拖拽

### 自定义节点组件

```typescript
// 每种节点类型对应一个组件
const nodeTypes = {
    trigger:   TriggerNode,
    rename:    RenameNode,
    compress:  CompressNode,
    thumbnail: ThumbnailNode,
    move:      MoveNode,
    condition: ConditionNode,
    wait:      WaitNode,
}

// 节点通用结构
interface BaseNodeData {
    label: string
    enabled: boolean
    // 执行状态（运行工作流时实时更新）
    status?: "idle" | "running" | "done" | "failed"
    errorMsg?: string
}

// 示例：RenameNode
function RenameNode({ data, selected }: NodeProps<RenameNodeData>) {
    return (
        <div className={cn("node-card", selected && "ring-2 ring-primary",
                           data.status === "done" && "border-green-500",
                           data.status === "failed" && "border-red-500")}>
            <Handle type="target" position={Position.Top} />
            <div className="node-header">
                <FileTextIcon /> <span>{data.label}</span>
                {data.status === "running" && <Spinner />}
            </div>
            <div className="node-body">
                <span className="text-muted">规则：</span>
                <span>{data.rule_name || "未配置"}</span>
            </div>
            <Handle type="source" position={Position.Bottom} />
        </div>
    )
}
```

### 执行状态可视化

运行工作流时，通过 SSE 实时更新节点状态：
- `workflow.node.start` → 节点显示旋转加载图标
- `workflow.node.done` → 节点边框变绿，显示耗时
- `workflow.node.failed` → 节点边框变红，显示错误信息
- 连线动画：当前执行路径高亮显示
